CREATE DATABASE [IF NOT EXISTS] race_war;

CREATE TABLE weapons (weapon_id integer,weapon_name varchar(20),weapon_image_path varchar(200),weapon_race varchar(20),weapon_points integer);

CREATE TABLE warriors (warrior_id integer,warrior_name varchar(30),warrior_image_path varchar(200),warrior_race varchar(5));

CREATE TABLE players (player_id integer,player_name varchar(30),damage_caused integer,damage_recieved integer,battle_points integer);

CREATE TABLE battle (battle_id integer,player_id integer,warrior_id integer,warrior_weapon_id integer,opponent_id integer,opponent_weapon_id integer,damage_caused integer,damage_recieved integer,battle_points integer);

CREATE TABLE races (race_name varchar(5),base_hp integer,base_strength integer,base_defense integer,base_dexterity integer,base_speed integer,points integer);
